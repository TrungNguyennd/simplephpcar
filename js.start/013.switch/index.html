câu lệnh switch case

    biểu thức thường là 1 biên trong js
    switch(biểu thức){
        case x:
        //các câu lệnh case x
        break;
        case y:
        //các câu lệnh case y
        break;
        case z:
        //các câu lệnh case z
        break;
    default:
    // các câu lệnh trong delault
    }
    biểu thức trong ()==x thì code block x sẽ chạy
    nếu biểu thức trong ()==y thì code block y sẽ chạy
    nếu biểu thức trong ()==z thì code block z sẽ chạy

    nếu trong trường hợp biểu thức trong () không bằng bất cứ 1 case nào
    thì code trong default sẽ chạy
